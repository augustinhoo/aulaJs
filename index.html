Curso Básico de JavaScript

Apresentação dos conceitos fundamentais — exemplos práticos em cada tópico

Total de tópicos: 14

JavaScript é a linguagem que torna páginas web interativas — manipula conteúdo, eventos e comportamento no navegador.

<script>
console.log("Olá, mundo!");
</script>

<script>
alert("Bem-vindo ao meu site!");
</script>

Inclusão inline ou via arquivo externo.

<!-- Exemplo 1: inline -->
<script>
console.log("Código interno executando...");
</script>

<!-- Exemplo 2: externo -->
<script src="meu-script.js"></script>

Armazenam dados — `var` (antigo), `let` (bloco), `const` (constante).

// Exemplo 1
let nome = "Ana";
console.log(nome);

// Exemplo 2
const idade = 25;
console.log("Idade:", idade);

Mostra mensagens e coleta entrada simples com `alert`, `prompt` e `confirm`.

// Exemplo 1 - alerta
alert("Olá, seja bem-vindo!");

// Exemplo 2 - prompt e exibição
let nome = prompt("Qual é o seu nome?");
alert("Prazer em te conhecer, " + nome + "!");

+, -, *, /, %, ** são usados para cálculos.

// Exemplo 1
let total = 10 + 5 * 2;
console.log(total); // 20

// Exemplo 2
let resto = 7 % 3;
console.log("Resto da divisão:", resto); // 1

==, ===, !=, >, <, >=, <=, &&, ||, ! — para comparar e combinar condições.

// Exemplo 1
let idade = 18;
if (idade >= 18 && idade <= 60) {
  console.log("Adulto");
}

// Exemplo 2
let temCarteira = true;
if (idade >= 18 && temCarteira) {
  console.log("Pode dirigir");
}

Executam blocos de código conforme condições.

// Exemplo 1 - if/else
let nota = 8;
if (nota >= 7) {
  console.log("Aprovado");
} else {
  console.log("Reprovado");
}

// Exemplo 2 - switch
let cor = "vermelho";
switch (cor) {
  case "azul":
    console.log("Cor fria");
    break;
  case "vermelho":
    console.log("Cor quente");
    break;
}

Repetem ações — `for`, `while`, `do...while`.

// Exemplo 1 - for
for (let i = 1; i <= 5; i++) {
  console.log("Número:", i);
}

// Exemplo 2 - while
let contador = 0;
while (contador < 3) {
  console.log("Contagem:", contador);
  contador++;
}

Acessa e altera elementos HTML dinamicamente.

<p id="texto">Olá!</p>
<button onclick="mudarTexto()">Clique</button>

<script>
function mudarTexto() {
  document.getElementById("texto").innerText = "Texto alterado!";
}
document.getElementById("texto").style.color = "blue";
</script>

Blocos reutilizáveis de código; recebem parâmetros e podem retornar valores.

// Exemplo 1 - função simples
function saudacao() {
  console.log("Olá!");
}
saudacao();

// Exemplo 2 - função com parâmetro e retorno
function somar(a, b) {
  return a + b;
}
console.log(somar(5, 3)); // 8

Coleções ordenadas — métodos úteis: push, pop, map, filter, etc.

// Exemplo 1 - acessar índice
let frutas = ["maçã", "banana", "uva"];
console.log(frutas[1]); // banana

// Exemplo 2 - adicionar item
frutas.push("laranja");
console.log(frutas);

Estrutura chave:valor para agrupar dados relacionados.

// Exemplo 1
let pessoa = { nome: "Carlos", idade: 30 };
console.log(pessoa.nome);

// Exemplo 2 - adicionar propriedade
pessoa.cidade = "São Paulo";
console.log(pessoa);

Reagem a ações do usuário: `click`, `keyup`, `submit`, `load`, etc.

<!-- Exemplo 1: inline -->
<button onclick="alert('Você clicou!')">Clique aqui</button>

<!-- Exemplo 2: addEventListener -->
<input type="text" id="campo">
<script>
document.getElementById('campo').addEventListener('keyup', function() {
  console.log('Digitou algo!');
});
</script>
